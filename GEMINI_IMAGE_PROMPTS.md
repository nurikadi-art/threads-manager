## BATCH GENERATION SCRIPT

```python
import google.generativeai as genai
import os
import time

genai.configure(api_key=os.environ.get("GEMINI_API_KEY"))
model = genai.ImageGenerationModel("imagen-3.0-generate-001")

# Style suffix to add to all prompts
STYLE_SUFFIX = """

Technical specifications:
- Photorealistic, shot on professional camera or iPhone 15 Pro
- NOT a stock photo, NOT AI-looking, NOT illustration
- Natural lighting, real textures, authentic environments
- Cinematic color grading: slightly desaturated, warm highlights, cool shadows
- 8K resolution, sharp focus on subject
"""

prompts = {
    "compass-luxury": '''[PASTE COMPASS PROMPT HERE]''',
    "founder-overwhelmed": '''[PASTE DATA BLINDNESS PROMPT HERE]''',
    "driving-idea": '''[PASTE LOST IDEA PROMPT HERE]''',
    "monday-inbox": '''[PASTE MONDAY PANIC PROMPT HERE]''',
    "chaos-clarity": '''[PASTE TRANSFORMATION PROMPT HERE]''',
    "hitl-workflow": '''[PASTE HITL PROMPT HERE]''',
    "sales-standup": '''[PASTE LEAD SCORING PROMPT HERE]''',
    "voice-recording": '''[PASTE CONTENT CREATION PROMPT HERE]''',
    "meeting-followup": '''[PASTE MEETING PROMPT HERE]''',
    "ai-operator-pro": '''[PASTE AI OPERATOR PROMPT HERE]''',
    "data-vault-leak": '''[PASTE DATA PRIVACY PROMPT HERE]''',
    "luxury-giftbox": '''[PASTE GIFT BOX PROMPT HERE]''',
    "onboarding-kit": '''[PASTE PROCESS PROMPT HERE]''',
    "testimonial-portrait": '''[PASTE TESTIMONIAL PROMPT HERE]''',
    "glitchy-error": '''[PASTE GLITCHY AI PROMPT HERE]''',
}

for name, prompt in prompts.items():
    print(f"\n{'='*50}")
    print(f"Generating: {name}")
    print(f"{'='*50}")

    try:
        full_prompt = prompt + STYLE_SUFFIX
        result = model.generate_images(
            prompt=full_prompt,
            number_of_images=2,  # Generate 2 variations to pick best
            aspect_ratio="16:9"
        )

        for i, img in enumerate(result.images):
            filename = f"{name}_v{i+1}.png"
            img.save(filename)
            print(f"Saved: {filename}")

        time.sleep(2)  # Rate limiting

    except Exception as e:
        print(f"Error generating {name}: {e}")
        continue

print("\n" + "="*50)
print("GENERATION COMPLETE!")
print("="*50)
```
